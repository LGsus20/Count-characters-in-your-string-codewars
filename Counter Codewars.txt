def print_hi(string):
    dictionary = {}
    for letter in string:
        if letter in dictionary:
            dictionary_number = dictionary.get(letter) + 1
            dictionary.update({letter: dictionary_number})
        else:
            dictionary[letter] = 1

    return dictionary

if __name__ == '__main__':
    inside_function ="teeest"
    print(print_hi((inside_function)))
